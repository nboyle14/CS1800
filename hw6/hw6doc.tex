\item 
$T(n) = T(\frac{9n}{10}) + n$  \\    %  \hskip1in      $T(1) = 1$\\
{\color{blue} Answer:  \\
(a)
\begin{multicols}{2}
$T(n) = T(\frac{9n}{10}) + n$ \\\\
$T(\frac{9n}{10}) = T(\frac{9^2n}{10^2}) + \frac{9n}{10}$  \\\\
$T(\frac{9^2n}{10^2}) = T(\frac{9^3n}{10^3}) + \frac{9^2n}{10^2}$  \\\\
$\hspace{3cm} ...$ \\\\
$T(\frac{9^{k-1}n}{10^{k-1}}) = T(\frac{9^{k}n}{10^k}) + \frac{9^{k-1}n}{10^{k-1}}$ \\\\
$T(\frac{9^{k}n}{10^{k}}) = T(1) $
\columnbreak 
\begin{center}
$ n $\\
$\downarrow$ \\ 
$\frac{9n}{10}$\\
$\downarrow$ \\ 
$\frac{9^2n}{10^2}$ \\
$\downarrow$ \\
...\\
$\downarrow$\\
$\frac{9^{k-1}n}{10^{k-1}}$ \\
$\downarrow$\\
$1$\\
\end{center}
\end{multicols}
From the last equation, we have $\frac{9^{k}}{10^{k}} \cdot n = 1$\\
$n = (\frac{10}{9})^k$ \\
$k = \frac{\log n }{\log (\frac{10}{9})}$ \\
$k = \log_{(\frac{10}{9})} n$ \hspace{2cm} Thus, the depth of the recursion is $\log_{(\frac{10}{9})} n$ \\
The total cost of the recursion is the sum of the cost at each level for the depth of the recursion, which is $\log_{(\frac{10}{9})} n$. \\
Thus, the total cost is given by: $\sum^{\log_{(\frac{10}{9})}n}_{k{=}0} (\frac{9}{10})^k \cdot n$  \\\\
$\sum^{\log_{(\frac{10}{9})}n}_{k{=}0} (\frac{9}{10})^k \cdot n = n \cdot \frac{1-(\frac{9}{10})^{\log_{\frac{10}{9}}n+1}}{1-\frac{9}{10}}$\\
$n \cdot \frac{1-(\frac{9}{10})^{\log_{\frac{10}{9}}n+1}}{1-\frac{9}{10}} = n\cdot\frac{1-\frac{1}{n}\cdot\frac{9}{10}}{\frac{1}{10}}$\\
$T(n) = 10n - 9$\\
Thus, $T(n) = \Theta(n)$$\blacksquare$\\

\newpage

(b)\\

$T(1) = 1$ \\\\
$T(2) = T(\frac{18}{10}) + 2 = 3.8$ \\\\
$T(3) = T(\frac{27}{10}) + 3 = 5.7$ \\\\
$T(100) = T(\frac{900}{10}) + 100 = 190$ \\\\
This begins to look like $2n$ \\\\
Hypothesis: $T(k) \leq ck$ for all $k < n$\\\\
Substitute: $T(n) \leq c \cdot (\frac{9n}{10}) + n$\\\\
$T(n) \leq n(0.9c+1)$\\\\
This is the desired form of $n \cdot c$\\\\
$0.9c + 1 \leq c$ \hspace{2cm} for all $c\geq10$ \\\\
Thus, $T(n) \leq 10n$ \hspace{2cm} for all $c\geq10$\\\\
Therefore, $T(n) = \Theta(n)$$\blacksquare$\\\\

}

\item  $S(n) = 9S(\frac{n}{3}) + n^2$     %  \hskip1in. $S(1) = 1$


{\color{blue} Answer:  \\

(a)   

\begin{table}[h]
    \centering
    \begin{tabular}{|c|c|c|c|}
        \hline
        \textbf{Recursive call} & \textbf{Tree Element} & \textbf{Number of Elements} & \textbf{Sum of Tree Elements} \\
        \hline
        $S(n) = 9S(\frac{n}{3})+n^2$ & $n^2$ & $1$ & $n^2$ \\
        \hline
        $S(\frac{n}{3}) = 9S(\frac{n}{3^2})+(\frac{n}{3})^2$ & $(\frac{n^2}{9})$ & $9$ & $n^2$ \\
        \hline
        $S(\frac{n}{3^2}) = 9S(\frac{n}{3^3})+(\frac{n}{3^2})^2$ & $(\frac{n^2}{81})$ & $81$ & $n^2$ \\
        \hline
        $S(\frac{n}{3^{k-1}}) = 9S(\frac{n}{3^{k}})+(\frac{n}{3^{k-1}})^2$ & $(\frac{n^2}{3^{k-1}})$  & $3^{k-1}$ & $n^2$ \\
        \hline
    \end{tabular}
    \label{tab:sample}
\end{table}

$S(1) = \frac{n^2}{3^{2k}} = 1$\\
$3^{2k} = n^2$\\
$k\log3 = \log n$\\
$k = \log_3n$\\
The depth, k, is equal to $\log_3n$ which means $n^2$ is summed $\log_3n$ times, i.e. $n^2 \cdot \log_3n$\\
Therefore, $S(n) = \Theta(n^2\log n)$$\blacksquare$\\

\vskip 1cm

(b)\\

Inductive Hypothesis: $S(k) \leq ck^2\log_3 k$ for all $k < n$\\
Substitute: $S(n) \leq 9c(\frac{n}{3})^2\log_3(\frac{n}{3}) + n^2$\\
$S(n) \leq cn^2\log_3(\frac{n}{3}) + n^2$\\
$S(n) \leq cn^2\log_3 n - cn^2\log_3 3 + n^2$\\
$S(n) \leq cn^2\log_3 n -(cn^2 -n^2)$\\
$(cn^2 -n^2) \geq 0$ \hspace{2cm} for all $c\geq1$\\
Thus, $S(n) \leq cn^2\log_3 n$ \hspace{2cm} for all $c\geq1$\\
Therefore, $S(n) = \Theta(n^2\log n)$$\blacksquare$\\
}






\end{enumerate}
